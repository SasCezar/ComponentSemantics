transform: translate(0px);",\n    1: {\n      "left": "100px",\n      "top": "0px",\n      transform: "translate(50px)",\n    },\n    2: {\n      "left": "200px",\n      "top": "100px",\n      transform: {\n        translate: "100px",\n      },\n    }\n  }\n}, {\n  selector: true,\n  easing: "ease-in-out",\n}).play();\n\n```\n## \xe2\x9c\xa8 Effects\n\n*
