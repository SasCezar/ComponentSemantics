errors } = useForm(); // initialise the hook\n  const onSubmit = data => {\n    console.log(data);\n  };\n\n  return (\n    <form onSubmit={handleSubmit(onSubmit)}>\n      <input name="firstname" ref={register} /> {/* register an input */}\n\n      <input name="lastname" ref={register({ required: true })} />\n      {errors.lastname && \'Last name is required.\'}\n\n      <input name="age" ref={register({ pattern: /\\d+/ })} />\n      {errors.age && \'Please enter number for age.\'}\n\n      <input type="submit" />\n    </form>\n  );\n}\n```\n\n## Contributors\n\nThanks goes to these
