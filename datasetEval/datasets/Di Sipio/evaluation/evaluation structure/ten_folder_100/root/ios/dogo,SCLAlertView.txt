Fluent style\n\n```Objective-C\n\nSCLAlertViewBuilder *builder = [SCLAlertViewBuilder new]\n.addButtonWithActionBlock(@"Send", ^{ /*work here*/ });\nSCLAlertViewShowBuilder *showBuilder = [SCLAlertViewShowBuilder new]\n.style(SCLAlertViewStyleWarning)\n.title(@"Title")\n.subTitle(@"Subtitle")\n.duration(0);\n[showBuilder showAlertView:builder.alertView onViewController:self.window.rootViewController];\n// or even\nshowBuilder.show(builder.alertView, self.window.rootViewController);\n```\n\n#### Complex\n```Objective-C\n    NSString *title = @"Title";\n    NSString *message = @"Message";\n    NSString *cancel = @"Cancel";\n    NSString *done = @"Done";\n    \n    SCLALertViewTextFieldBuilder *textField = [SCLALertViewTextFieldBuilder new].title(@"Code");\n    SCLALertViewButtonBuilder *doneButton = [SCLALertViewButtonBuilder new].title(done)\n    .validationBlock(^BOOL{\n        NSString *code = [textField.textField.text copy];\n        return [code isVisible];\n    })\n    .actionBlock(^{\n        NSString *code = [textField.textField.text copy];\n        [self confirmPhoneNumberWithCode:code];\n    });\n    \n    SCLAlertViewBuilder *builder = [SCLAlertViewBuilder new]\n    .showAnimationType(SCLAlertViewShowAnimationFadeIn)\n    .hideAnimationType(SCLAlertViewHideAnimationFadeOut)\n    .shouldDismissOnTapOutside(NO)\n    .addTextFieldWithBuilder(textField)\n    .addButtonWithBuilder(doneButton);\n    \n    SCLAlertViewShowBuilder *showBuilder = [SCLAlertViewShowBuilder new]\n    .style(SCLAlertViewStyleCustom)\n    .image([SCLAlertViewStyleKit imageOfInfo])\n    .color([UIColor blueColor])\n    .title(title)\n    .subTitle(message)\n    .closeButtonTitle(cancel)\n    .duration(0.0f);\n\n    [showBuilder showAlertView:builder.alertView onViewController:self];\n```\n\n### Easy to use\n```Objective-C\n// Get started\nSCLAlertView *alert = [[SCLAlertView alloc] init];\n\n[alert showSuccess:self title:@"Hello World" subTitle:@"This is a more descriptive text." closeButtonTitle:@"Done" duration:0.0f];\n\n// Alternative alert types\n[alert showError:self title:@"Hello Error" subTitle:@"This is a more descriptive error text." closeButtonTitle:@"OK" duration:0.0f]; // Error\n[alert showNotice:self title:@"Hello Notice" subTitle:@"This is a more descriptive notice text." closeButtonTitle:@"Done" duration:0.0f]; // Notice\n[alert showWarning:self title:@"Hello Warning" subTitle:@"This is a more descriptive warning text." closeButtonTitle:@"Done" duration:0.0f]; // Warning\n[alert showInfo:self title:@"Hello Info" subTitle:@"This is a more descriptive info text." closeButtonTitle:@"Done" duration:0.0f]; // Info\n[alert showEdit:self title:@"Hello Edit" subTitle:@"This is a more descriptive info text with a edit textbox" closeButtonTitle:@"Done" duration:0.0f]; // Edit\n[alert showCustom:self image:[UIImage imageNamed:@"git"] color:color title:@"Custom" subTitle:@"Add a custom icon and color for your own type of alert!" closeButtonTitle:@"OK" duration:0.0f]; // Custom\n[alert showWaiting:self title:@"Waiting..." subTitle:@"Blah de blah de blah, blah. Blah de blah de" closeButtonTitle:nil duration:5.0f];\n[alert showQuestion:self title:@"Question?" subTitle:kSubtitle closeButtonTitle:@"Dismiss" duration:0.0f];\n\n\n// Using custom alert width\nSCLAlertView *alert = [[SCLAlertView alloc] initWithWindowWidth:300.0f];\n```\n\n### SCLAlertview in a new window. (No UIViewController)\n```Objective-C\n\nSCLAlertView *alert = [[SCLAlertView alloc] initWithNewWindow];\n\n[alert showSuccess:@"Hello World" subTitle:@"This is a more descriptive text." closeButtonTitle:@"Done" duration:0.0f];\n\n// Alternative alert types\n[alert showError:@"Hello Error" subTitle:@"This is a more descriptive error text." closeButtonTitle:@"OK" duration:0.0f]; // Error\n[alert showNotice:@"Hello Notice" subTitle:@"This is a more descriptive notice text." closeButtonTitle:@"Done" duration:0.0f]; // Notice\n[alert showWarning:@"Hello Warning" subTitle:@"This is a more descriptive warning text." closeButtonTitle:@"Done" duration:0.0f]; // Warning\n[alert showInfo:@"Hello Info" subTitle:@"This is a more descriptive info text." closeButtonTitle:@"Done" duration:0.0f]; // Info\n[alert showEdit:@"Hello Edit" subTitle:@"This is a more descriptive info text with a edit textbox" closeButtonTitle:@"Done" duration:0.0f]; // Edit\n[alert showCustom:[UIImage imageNamed:@"git"] color:color title:@"Custom" subTitle:@"Add a custom icon and color for your own type of alert!" closeButtonTitle:@"OK" duration:0.0f]; // Custom\n[alert showWaiting:@"Waiting..." subTitle:@"Blah de blah de blah, blah. Blah de blah de" closeButtonTitle:nil duration:5.0f];\n[alert showQuestion:@"Question?" subTitle:kSubtitle closeButtonTitle:@"Dismiss" duration:0.0f];\n\n// Using custom alert width\nSCLAlertView *alert = [[SCLAlertView alloc] initWithNewWindowWidth:300.0f];\n```\n\n### New Window: Known issues\n\n1. SCLAlert animation is wrong in landscape. (iOS 6.X and 7.X)\n\n### Add buttons\n```Objective-C\nSCLAlertView *alert = [[SCLAlertView alloc] init];\n\n//Using Selector\n[alert addButton:@"First Button" target:self selector:@selector(firstButton)];\n\n//Using Block\n[alert addButton:@"Second Button" actionBlock:^(void) {\n    NSLog(@"Second button tapped");\n}];\n\n//Using Blocks With Validation\n[alert addButton:@"Validate" validationBlock:^BOOL {\n    BOOL passedValidation = ....\n    return passedValidation;\n\n} actionBlock:^{\n    // handle successful validation here\n}];\n\n[alert showSuccess:self title:@"Button View" subTitle:@"This alert view has buttons" closeButtonTitle:@"Done" duration:0.0f];\n```\n\n### Add button timer\n```Objective-C\n//The index of the button to add the timer display to.\n[alert addTimerToButtonIndex:0 reverse:NO];\n```\n\nExample:\n\n```Objective-C\nSCLAlertView *alert = [[SCLAlertView alloc] init];\n[alert addTimerToButtonIndex:0 reverse:YES];\n[alert showInfo:self title:@"Countdown Timer" subTitle:@"This alert has a duration set, and a countdown timer on the Dismiss button to show how long is left." closeButtonTitle:@"Dismiss" duration:10.0f];\n```\n\n\n### Add Text Attributes\n```Objective-C\nSCLAlertView *alert = [[SCLAlertView alloc] init];\n\nalert.attributedFormatBlock = ^NSAttributedString* (NSString *value)\n{\n    NSMutableAttributedString *subTitle = [[NSMutableAttributedString alloc]initWithString:value];\n\n    NSRange redRange = [value rangeOfString:@"Attributed" options:NSCaseInsensitiveSearch];\n    [subTitle addAttribute:NSForegroundColorAttributeName value:[UIColor redColor] range:redRange];\n\n    NSRange greenRange = [value rangeOfString:@"successfully" options:NSCaseInsensitiveSearch];\n    [subTitle addAttribute:NSForegroundColorAttributeName value:[UIColor greenColor] range:greenRange];\n\n    NSRange underline = [value rangeOfString:@"completed" options:NSCaseInsensitiveSearch];\n    [subTitle addAttributes:@{NSUnderlineStyleAttributeName:@(NSUnderlineStyleSingle)} range:underline];\n\n    return subTitle;\n};\n\n[alert showSuccess:self title:@"Button View" subTitle:@"Attributed string operation successfully completed." closeButtonTitle:@"Done" duration:0.0f];\n```\n\n### Add a text field\n```Objective-C\nSCLAlertView *alert = [[SCLAlertView alloc] init];\n\nUITextField *textField = [alert addTextField:@"Enter your name"];\n\n[alert addButton:@"Show Name" actionBlock:^(void) {\n    NSLog(@"Text value: %@", textField.text);\n}];\n\n[alert showEdit:self title:@"Edit View" subTitle:@"This alert view shows a text box" closeButtonTitle:@"Done" duration:0.0f];\n```\n\n### Indeterminate progress\n```Objective-C\nSCLAlertView *alert = [[SCLAlertView alloc] init];\n    \n[alert showWaiting:self title:@"Waiting..." subTitle:@"Blah de blah de blah, blah. Blah de blah de" closeButtonTitle:nil duration:5.0f];\n```\n\n### Add a switch button\n```Objective-C\nSCLAlertView *alert = [[SCLAlertView alloc] init];\n    \nSCLSwitchView *switchView = [alert addSwitchViewWithLabel:@"Don\'t show again".uppercaseString];\nswitchView.tintColor = [UIColor brownColor];\n    \n[alert addButton:@"Done" actionBlock:^(void) {\n    NSLog(@"Show again? %@", switchView.isSelected ? @"-No": @"-Yes");\n}];\n    \n[alert showCustom:self image:[UIImage imageNamed:@"switch"] color:[UIColor brownColor] title:kInfoTitle subTitle:kSubtitle closeButtonTitle:nil duration:0.0f];\n```\n\n### Add custom view\n```Objective-C\nSCLAlertView *alert = [[SCLAlertView alloc] init];\n\nUIView *customView = [[UIView alloc] initWithFrame:CGRectMake(0.0f, 0.0f, 215.0f, 80.0f)];\ncustomView.backgroundColor = [UIColor redColor];\n\n[alert addCustomView:customView];\n\n[alert showNotice:self title:@"Title" subTitle:@"This alert view shows a custom view" closeButtonTitle:@"Done" duration:0.0f];\n```\n\n### SCLAlertView properties\n```Objective-C\n//Dismiss on tap outside (Default is NO)\nalert.shouldDismissOnTapOutside = YES;\n\n//Hide animation type (Default is SCLAlertViewHideAnimationFadeOut)\nalert.hideAnimationType = SCLAlertViewHideAnimationSlideOutToBottom;\n\n//Show animation type (Default is SCLAlertViewShowAnimationSlideInFromTop)\nalert.showAnimationType =  SCLAlertViewShowAnimationSlideInFromLeft;\n\n//Set background type (Default is SCLAlertViewBackgroundShadow)\nalert.backgroundType = SCLAlertViewBackgroundBlur;\n\n//Overwrite SCLAlertView (Buttons, top circle and borders) colors\nalert.customViewColor = [UIColor purpleColor];\n\n//Set custom tint color for icon image.\nalert.iconTintColor = [UIColor purpleColor];\n\n//Override top circle tint color with background color\nalert.tintTopCircle = NO;\n\n//Set custom corner radius for SCLAlertView\nalert.cornerRadius = 13.0f;\n\n//Overwrite SCLAlertView background color\nalert.backgroundViewColor = [UIColor cyanColor];\n\n//Returns if the alert is visible or not.\nalert.isVisible;\n\n//Make the top circle icon larger\nalert.useLargerIcon = YES;\n\n//Using sound\nalert.soundURL = [NSURL fileURLWithPath:[NSString stringWithFormat:@"%@/right_answer.mp3", [[NSBundle mainBundle] resourcePath]]];\n\n\n```\n\n### Helpers\n```Objective-C\n//Receiving information that SCLAlertView is dismissed\n[alert alertIsDismissed:^{\n    NSLog(@"SCLAlertView
